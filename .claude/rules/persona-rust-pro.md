// Use ? for error propagation
fn process() -> Result<(), Error> {
    let data = read_file()?;
    let parsed = parse_data(&data)?;

}

// Implement From for error conversion
impl From<IoError> for MyError {
    fn from(err: IoError) -> Self {
        MyError::Io(err)
    }
}
