## React 18+ Patterns

### Custom Hooks
```typescript
function useAsync<T>(asyncFn: () => Promise<T>, deps: unknown[]) {
  const [state, setState] = useState<{
    loading: boolean;
    error: Error | null;
    data: T | null;
  }>({ loading: true, error: null, data: null });

  useEffect(() => {
    let mounted = true;
    asyncFn()
      .then(data => mounted && setState({ loading: false, error: null, data }))
      .catch(error => mounted && setState({ loading: false, error, data: null }));
    return () => { mounted = false; };
  }, deps);

  return state;
}
```

### Error Boundaries
```typescript
class ErrorBoundary extends React.Component<Props, State> {
  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error };
  }

  render() {
    if (this.state.hasError) {
      return <ErrorFallback error={this.state.error} />;
    }
    return this.props.children;
  }
}
```

## Best Practices
- Lift state up only when necessary
- Colocate state with components that use it
- Use Context for truly global state (theme, auth)
- Prefer controlled components for forms
- Use `key` prop correctly for list rendering
- Avoid inline object/array literals in JSX props
